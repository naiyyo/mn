Pomerol :

conf term
version 12.2
hostname Pomerol
ip subnet-zero
ip cef

interface Loopback0
ip address 10.10.10.3 255.255.255.255

interface Serial4/0
ip address 10.1.1.21 255.255.255.252
tag-switching ip

interface Serial4/2
ip address 10.1.1.6 255.255.255.252
tag-switching ip

interface Serial4/4
ip address 10.1.1.9 255.255.255.252
tag-switching ip

router ospf 10
log-adjacency-changes
network 10.0.0.0 0.255.255.255 area 9

ip classless

end
